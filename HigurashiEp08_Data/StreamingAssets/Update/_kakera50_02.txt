void main()
{
	DrawScene( "black", 400 );

	DrawScene( "white", 1000 );

	PlayBGM( 0, "maturi_01_Ver2", 120, 0 );

	DrawSceneWithMask( "background/fure2", "maskm1", 1, 0, 1300 );

	ClearMessage();
	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　……長かった。",
		   NULL, "Demorou muito.", Line_WaitForInput);
	OutputLine(NULL, "…本当に長かった。",
		   NULL, " Realmente demorou muito.", Line_Normal);
	ClearMessage();

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　大丈夫…？",
		   NULL, "Você está bem?", Line_WaitForInput);
	OutputLine(NULL, "　まだ、あなたの精神は生きている…？",
		   NULL, " Seu espírito ainda está vivo?", Line_Normal);
	ClearMessage();

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　これで、私たちが欲しい駒は全てゲーム盤に並んだのかしら。",
		   NULL, "Temos todas as peças que queremos alinhadas no tabuleiro do jogo?", GetGlobalFlag(GLinemodeSp));
	if (GetGlobalFlag(GADVMode)) { ClearMessage(); } else { OutputLineAll(NULL, "\n", Line_ContinueAfterTyping); }

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　昭和５８年６月を巡る、運命のゲームで、鷹野たちに勝てるだけの駒が並んだのかしら。",
		   NULL, "Isso é tudo o que temos que para enfrentar Takano e seus subordinados no jogo do destino em junho de 1983?", Line_Normal);
	ClearMessage();

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　……私は、",
		   NULL, "Eu acho que,", Line_WaitForInput);
	OutputLine(NULL, "多分、",
		   NULL, " talvez,", Line_ContinueAfterTyping);
	SetValidityOfInput( FALSE );
	Wait( 800 );
	SetValidityOfInput( TRUE );
	OutputLine(NULL, "並んだと思ってる。",
		   NULL, " nós temos tudo.", Line_Normal);
	ClearMessage();

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　あとは、羽入。",
		   NULL, "E por fim, mas não menos importante, Hanyuu,", Line_WaitForInput);
	OutputLine(NULL, "…あなただけね。",
		   NULL, " é você.", GetGlobalFlag(GLinemodeSp));
	if (GetGlobalFlag(GADVMode)) { ClearMessage(); } else { OutputLineAll(NULL, "\n", Line_ContinueAfterTyping); }

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　あなたが駒として登場できる時間は、もう相当に制限されている。",
		   NULL, "O tempo que lhe resta para ser uma peça é limitado.", GetGlobalFlag(GLinemodeSp));
	if (GetGlobalFlag(GADVMode)) { ClearMessage(); } else { OutputLineAll(NULL, "\n", Line_ContinueAfterTyping); }

	if (GetGlobalFlag(GADVMode) == 0) { OutputLineAll(NULL, "\n", Line_ContinueAfterTyping); }


	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　でも、ゲームに登場できなくなったわけじゃない。",
		   NULL, "Mas isso não significa que você não possa aparecer no jogo.", Line_Normal);
	ClearMessage();

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　………さぁ、始めましょう？",
		   NULL, "...Vamos começar?", GetGlobalFlag(GLinemodeSp));
	if (GetGlobalFlag(GADVMode)) { ClearMessage(); } else { OutputLineAll(NULL, "\n", Line_ContinueAfterTyping); }

	if (GetGlobalFlag(GADVMode)) { OutputLineAll("", NULL, Line_ContinueAfterTyping); }
	OutputLine(NULL, "　この日のために、数百年もの時間を賭けたのだから。",
		   NULL, "Eu dediquei centenas de anos só para este momento.", Line_Normal);
	ClearMessage();

	DisableWindow();
	SetValidityOfInput( FALSE );
	Wait( 5000 );
	SetValidityOfInput( TRUE );
	CallScript( "&eye" );
	FadeOutBGM( 0, 1000, FALSE );
	DrawScene( "black", 1000 );

}
